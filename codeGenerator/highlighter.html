Binary and Octal Literals 


// ES 5 Code
var myBinary = 01001;
console.log(myBinary); // 513



// ES 2015 Code
var myBinary = 0b01001;
console.log(myBinary); // 9



// ES 5 Code
var myOctal = 0170;
console.log(myOctal); // 120



// ES 2015 Code
var myOctal = 0o170;
console.log(myOctal); // 120



var str = 'To be, or not to be, that is the question.';
console.log(str.startsWith('To be'));         // true
console.log(str.startsWith('not to be'));     // false



console.log(str.endsWith('question.')); // true
console.log(str.endsWith('to be'));     // false



'Blue Whale'.includes('blue'); // returns false (Case Sensitive)
console.log(str.includes('To be'));       // true
console.log(str.includes('question'));    // true
console.log(str.includes('nonexistent')); // false



'abc'.repeat(0);    // ''
'abc'.repeat(1);    // 'abc'
'abc'.repeat(2);    // 'abcabc'



String.fromCodePoint(42);       // "*"
String.fromCodePoint(65, 90);   // "AZ"



'ABC'.codePointAt(1);          // 66
'\uD800\uDC00'.codePointAt(0); // 65536

'XYZ'.codePointAt(42); // undefined



let myArr = [1,2,3];
for (let value of myArr) {
  console.log(`value = ${value}`); // Outputs 1 then 2 then 3
}



let myStr = 'the lazy brown dog';
for (let char of myStr) {
  console.log(`char = ${char}`); // Outputs each char
}



let myObj = {
  first: '1st',
  second: '2nd',
  last: '3rd'
}

myObj[Symbol.iterator] = function * () {
  yield 'second';
  yield 'last';
  yield 'first';
};

for (let key of myObj) {
  console.log(`key = ${key} -- ${myObj[key]}`); // outputs second then last then first
}



let iterable = [3, 5, 7];
iterable.foo = "hello";

for (let i in iterable) {
  console.log(i); // logs 0, 1, 2, "foo"
}

for (let i of iterable) {
  console.log(i); // logs 3, 5, 7
}
